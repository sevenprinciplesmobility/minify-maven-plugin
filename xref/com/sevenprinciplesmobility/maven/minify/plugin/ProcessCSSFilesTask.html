19  package com.sevenprinciplesmobility.maven.minify.plugin;
20  
21  import com.samaxes.maven.minify.common.YuiConfig;
22  import com.sevenprinciplesmobility.maven.minify.plugin.MinifyMojo.Engine;
23  import com.yahoo.platform.yui.compressor.CssCompressor;
24  import org.apache.maven.plugin.logging.Log;
25  
26  import java.io.*;
27  import java.nio.charset.Charset;
28  import java.util.List;
29  
30  /**
31   * Task for merging and compressing CSS files.
32   */
33  public class ProcessCSSFilesTask extends ProcessFilesTask {
34  
35      /**
36       * Task constructor.
37       *
38       * @param log             Maven plugin log
39       * @param verbose         display additional info
40       * @param bufferSize      size of the buffer used to read source files
41       * @param charset         if a character set is specified, a byte-to-char variant allows the encoding to be selected.
42       *                        Otherwise, only byte-to-byte operations are used
43       * @param suffix          final file name suffix
44       * @param nosuffix        whether to use a suffix for the minified file name or not
45       * @param skipMerge       whether to skip the merge step or not
46       * @param skipMinify      whether to skip the minify step or not
47       * @param webappSourceDir web resources source directory
48       * @param webappTargetDir web resources target directory
49       * @param inputDir        directory containing source files
50       * @param sourceFiles     list of source files to include
51       * @param sourceIncludes  list of source files to include
52       * @param sourceExcludes  list of source files to exclude
53       * @param outputDir       directory to write the final file
54       * @param outputFilename  the output file name
55       * @param engine          minify processor engine selected
56       * @param yuiConfig       YUI Compressor configuration
57       * @throws FileNotFoundException when the given source file does not exist
58       */
59      public ProcessCSSFilesTask(Log log, boolean verbose, Integer bufferSize, Charset charset, String suffix,
60                                 boolean nosuffix, boolean skipMerge, boolean skipMinify, String webappSourceDir,
61                                 String webappTargetDir, String inputDir, List<String> sourceFiles,
62                                 List<String> sourceIncludes, List<String> sourceExcludes, String outputDir,
63                                 String outputFilename, Engine engine, YuiConfig yuiConfig) throws FileNotFoundException {
64          super(log, verbose, bufferSize, charset, suffix, nosuffix, skipMerge, skipMinify, webappSourceDir,
65                  webappTargetDir, inputDir, sourceFiles, sourceIncludes, sourceExcludes, outputDir, outputFilename,
66                  engine, yuiConfig);
67      }
68  
69      /**
70       * Minifies a CSS file. Create missing parent directories if needed.
71       *
72       * @param mergedFile   input file resulting from the merged step
73       * @param minifiedFile output file resulting from the minify step
74       * @throws IOException when the minify step fails
75       */
76      @Override
77      protected void minify(File mergedFile, File minifiedFile) throws IOException {
78          if (!minifiedFile.getParentFile().exists() && !minifiedFile.getParentFile().mkdirs()) {
79              throw new RuntimeException("Unable to create target directory for: " + minifiedFile.getParentFile());
80          }
81  
82          try (InputStream in = new FileInputStream(mergedFile);
83               OutputStream out = new FileOutputStream(minifiedFile);
84               InputStreamReader reader = new InputStreamReader(in, charset);
85               OutputStreamWriter writer = new OutputStreamWriter(out, charset)) {
86              log.info("Creating the minified file [" + (verbose ? minifiedFile.getPath() : minifiedFile.getName())
87                      + "].");
88  
89              switch (engine) {
90                  case YUI:
91                      log.debug("Using YUI Compressor engine.");
92  
93                      CssCompressor compressor = new CssCompressor(reader);
94                      compressor.compress(writer, yuiConfig.getLineBreak());
95                      break;
96                  default:
97                      log.warn("CSS engine not supported.");
98                      break;
99              }
100         } catch (IOException e) {
101             log.error("Failed to compress the CSS file [" + (verbose ? mergedFile.getPath() : mergedFile.getName())
102                     + "].", e);
103             throw e;
104         }
105 
106         logCompressionGains(mergedFile, minifiedFile);
107     }
108 }
