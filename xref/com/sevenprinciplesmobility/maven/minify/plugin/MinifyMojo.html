444     /**
445      * Executed when the goal is invoked, it will first invoke a parallel lifecycle, ending at the given phase.
446      */
447     @Override
448     public void execute() throws MojoExecutionException, MojoFailureException {
449         if (skipMerge && skipMinify) {
450             getLog().warn("Both merge and minify steps are configured to be skipped.");
451             return;
452         }
453 
454         fillOptionalValues();
455 
456         YuiConfig yuiConfig = fillYuiConfig();
457         ClosureConfig closureConfig = fillClosureConfig();
458         Collection<ProcessFilesTask> processFilesTasks;
459         try {
460             processFilesTasks = createTasks(yuiConfig, closureConfig);
461         } catch (FileNotFoundException e) {
462             throw new MojoFailureException(e.getMessage(), e);
463         }
464 
465         ExecutorService executor = Executors.newFixedThreadPool(processFilesTasks.size());
466         try {
467             List<Future<Object>> futures = executor.invokeAll(processFilesTasks);
468             for (Future<Object> future : futures) {
469                 try {
470                     future.get();
471                 } catch (ExecutionException e) {
472                     throw new MojoExecutionException(e.getMessage(), e);
473                 }
474             }
475             executor.shutdown();
476         } catch (InterruptedException e) {
477             executor.shutdownNow();
478             throw new MojoExecutionException(e.getMessage(), e);
479         }
480     }

482     private void fillOptionalValues() {
483         if (Strings.isNullOrEmpty(cssTargetDir)) {
484             cssTargetDir = cssSourceDir;
485         }
486         if (Strings.isNullOrEmpty(jsTargetDir)) {
487             jsTargetDir = jsSourceDir;
488         }
489         if (Strings.isNullOrEmpty(charset)) {
490             charset = Charset.defaultCharset().name();
491         }
492     }

494     private YuiConfig fillYuiConfig() {
495         return new YuiConfig(yuiLineBreak, !yuiNoMunge, yuiPreserveSemicolons, yuiDisableOptimizations);
496     }

498     private ClosureConfig fillClosureConfig() throws MojoFailureException {
499         DependencyOptions dependencyOptions;
500         if(closureSortDependencies){
501             dependencyOptions = DependencyOptions.sortOnly();
502         }else{
503             dependencyOptions = DependencyOptions.none();
504         }
505 
506         List<SourceFile> externs = new ArrayList<>();
507         for (String extern : closureExterns) {
508             externs.add(SourceFile.fromFile(webappSourceDir + File.separator + extern, Charset.forName(charset)));
509         }
510 
511         Map<DiagnosticGroup, CheckLevel> warningLevels = new HashMap<>();
512         DiagnosticGroups diagnosticGroups = new DiagnosticGroups();
513         for (Map.Entry<String, String> warningLevel : closureWarningLevels.entrySet()) {
514             DiagnosticGroup diagnosticGroup = diagnosticGroups.forName(warningLevel.getKey());
515             if (diagnosticGroup == null) {
516                 throw new MojoFailureException("Failed to process closureWarningLevels: " + warningLevel.getKey() + " is an invalid DiagnosticGroup");
517             }
518 
519             try {
520                 CheckLevel checkLevel = CheckLevel.valueOf(warningLevel.getValue());
521                 warningLevels.put(diagnosticGroup, checkLevel);
522             } catch (IllegalArgumentException e) {
523                 throw new MojoFailureException("Failed to process closureWarningLevels: " + warningLevel.getKey() + " is an invalid CheckLevel");
524             }
525         }
526 
527         return new ClosureConfig(closureLanguageIn, closureLanguageOut, closureEnvironment, closureCompilationLevel,
528                 dependencyOptions, externs, closureCreateSourceMap, warningLevels, closureAngularPass,
529                 closureExtraAnnotations, closureDefine);
530     }

532     private Collection<ProcessFilesTask> createTasks(YuiConfig yuiConfig, ClosureConfig closureConfig)
533             throws MojoFailureException, FileNotFoundException {
534         List<ProcessFilesTask> tasks = new ArrayList();
535 
536         if (!Strings.isNullOrEmpty(bundleConfiguration)) { // If a bundleConfiguration is defined, attempt to use that
537             AggregationConfiguration aggregationConfiguration;
538             try (Reader bundleConfigurationReader = new FileReader(bundleConfiguration)) {
539                 aggregationConfiguration = new Gson().fromJson(bundleConfigurationReader,
540                         AggregationConfiguration.class);
541             } catch (IOException e) {
542                 throw new MojoFailureException("Failed to open the bundle configuration file [" + bundleConfiguration
543                         + "].", e);
544             }
545 
546             for (Aggregation aggregation : aggregationConfiguration.getBundles()) {
547                 if (Aggregation.AggregationType.css.equals(aggregation.getType())) {
548                     tasks.add(createCSSTask(yuiConfig, closureConfig, aggregation.getFiles(),
549                             Collections.<String>emptyList(), Collections.<String>emptyList(), aggregation.getName()));
550                 } else if (Aggregation.AggregationType.js.equals(aggregation.getType())) {
551                     tasks.add(createJSTask(yuiConfig, closureConfig, aggregation.getFiles(),
552                             Collections.<String>emptyList(), Collections.<String>emptyList(), aggregation.getName()));
553                 }
554             }
555         } else { // Otherwise, fallback to the default behavior
556             tasks.add(createCSSTask(yuiConfig, closureConfig, cssSourceFiles, cssSourceIncludes, cssSourceExcludes,
557                     cssFinalFile));
558             tasks.add(createJSTask(yuiConfig, closureConfig, jsSourceFiles, jsSourceIncludes, jsSourceExcludes,
559                     jsFinalFile));
560         }
561 
562         return tasks;
563     }
564 
565     private ProcessFilesTask createCSSTask(YuiConfig yuiConfig, ClosureConfig closureConfig,
566                                            List<String> cssSourceFiles, List<String> cssSourceIncludes, List<String> cssSourceExcludes,
567                                            String cssFinalFile) throws FileNotFoundException {
568         return new ProcessCSSFilesTask(getLog(), verbose, bufferSize, Charset.forName(charset), suffix, nosuffix,
569                 skipMerge, skipMinify, webappSourceDir, webappTargetDir, cssSourceDir, cssSourceFiles,
570                 cssSourceIncludes, cssSourceExcludes, cssTargetDir, cssFinalFile, cssEngine, yuiConfig);
571     }
572 
573     private ProcessFilesTask createJSTask(YuiConfig yuiConfig, ClosureConfig closureConfig, List<String> jsSourceFiles,
574                                           List<String> jsSourceIncludes, List<String> jsSourceExcludes, String jsFinalFile)
575             throws FileNotFoundException {
576         return new ProcessJSFilesTask(getLog(), verbose, bufferSize, Charset.forName(charset), suffix, nosuffix,
577                 skipMerge, skipMinify, webappSourceDir, webappTargetDir, jsSourceDir, jsSourceFiles, jsSourceIncludes,
578                 jsSourceExcludes, jsTargetDir, jsFinalFile, jsEngine, yuiConfig, closureConfig);
579     }
